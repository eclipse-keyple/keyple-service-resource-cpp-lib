79PluginsConfigurator::Builder& PluginsConfigurator::Builder::addPluginWithMonitoring(
80 std::shared_ptr<Plugin> plugin,
81 std::shared_ptr<ReaderConfiguratorSpi> readerConfiguratorSpi,
82 std::shared_ptr<PluginObservationExceptionHandlerSpi> pluginObservationExceptionHandlerSpi,
83 std::shared_ptr<CardReaderObservationExceptionHandlerSpi> readerObservationExceptionHandlerSpi)
84{
85 Assert::getInstance().notNull(plugin, "plugin")
86 .notNull(readerConfiguratorSpi, "readerConfiguratorSpi");
87
88 const auto pool = std::dynamic_pointer_cast<PoolPlugin>(plugin);
89 if (pool != nullptr) {
90 throw IllegalArgumentException("Plugin must be an instance of Plugin or ObservablePlugin");
91 }
92
93 if (Arrays::contains(mPlugins, plugin)) {
94 throw IllegalStateException("Plugin already configured.");
95 }
96
97 mPlugins.push_back(plugin);
98 mConfiguredPlugins.push_back(
99 std::make_shared<ConfiguredPlugin>(plugin,
100 readerConfiguratorSpi,
101 pluginObservationExceptionHandlerSpi,
102 readerObservationExceptionHandlerSpi));
103
104 return *this;
105}

107std::shared_ptr<PluginsConfigurator> PluginsConfigurator::Builder::build()
108{
109 if (mPlugins.empty()) {
110 throw IllegalStateException("No plugin was configured.");
111 }
112
113 if (mAllocationStrategyConfigured == false) {
114 mAllocationStrategy = AllocationStrategy::FIRST;
115 mAllocationStrategyConfigured = true;
116 }
117
118 if (mUsageTimeoutMillisConfigured == false) {
119 /* Infinite */
120 mUsageTimeoutMillis = 0;
121 mUsageTimeoutMillisConfigured = true;
122 }
123
124 return std::make_shared<PluginsConfigurator>(this);
125}
