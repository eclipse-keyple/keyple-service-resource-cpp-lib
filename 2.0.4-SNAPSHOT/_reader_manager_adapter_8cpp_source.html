74std::shared_ptr<CardResource> ReaderManagerAdapter::matches(
75 std::shared_ptr<CardResourceProfileExtension> extension)
76{
77 std::shared_ptr<CardResource> cardResource = nullptr;
78 std::shared_ptr<SmartCard> smartCard =
79 extension->matches(mReader,
80 SmartCardServiceProvider::getService()->createCardSelectionManager());

82 if (smartCard != nullptr) {

92bool ReaderManagerAdapter::lock(std::shared_ptr<CardResource> cardResource,
93 std::shared_ptr<CardResourceProfileExtension> extension)
94{
95 if (mIsBusy) {
96 if (static_cast<uint64_t>(System::currentTimeMillis()) < mLockMaxTimeMillis) {
97 return false;
98 }
99
100 mLogger->warn("Reader '%' automatically unlocked due to a usage duration over than % " \
101 "milliseconds\n",
102 mReader->getName(),
103 mUsageTimeoutMillis);
104 }
105
106 if (mSelectedCardResource != cardResource) {
107 std::shared_ptr<SmartCard> smartCard =
108 extension->matches(
109 mReader,
110 SmartCardServiceProvider::getService()->createCardSelectionManager());
111
112 if (!areEquals(cardResource->getSmartCard(), smartCard)) {
113 mSelectedCardResource = nullptr;
114 throw IllegalStateException("No card is inserted or its profile does not match the " \
115 "associated data.");
116 }
117
118 mSelectedCardResource = cardResource;
119 }
120
121 mLockMaxTimeMillis = System::currentTimeMillis() + mUsageTimeoutMillis;
122 mIsBusy = true;
123
124 return true;
125}

140std::shared_ptr<CardResource> ReaderManagerAdapter::getOrCreateCardResource(
141 std::shared_ptr<SmartCard> smartCard)
142{
143 /* Check if an identical card resource is already created */
144 for (const auto& cardResource : mCardResources) {
145 if (areEquals(cardResource->getSmartCard(), smartCard)) {
146 return cardResource;
147 }
148 }
149
150 /* If none, then create a new one */
151 auto cardResource = std::make_shared<CardResource>(mReader, smartCard);
152 mCardResources.push_back(cardResource);
153
154 return cardResource;
155}

157bool ReaderManagerAdapter::areEquals(const std::shared_ptr<SmartCard> s1,
158 const std::shared_ptr<SmartCard> s2) const
159{
160 if (s1 == s2) {
161 return true;
162 }
163
164 if (s1 == nullptr || s2 == nullptr) {
165 return false;
166 }
167
168 bool hasSamePowerOnData = (s1->getPowerOnData() == "" && s2->getPowerOnData() == "") ||
169 (s1->getPowerOnData() != "" &&
170 s1->getPowerOnData() == s2->getPowerOnData());
171
172 bool hasSameFci = Arrays::equals(s1->getSelectApplicationResponse(),
173 s2->getSelectApplicationResponse());
174
175 return hasSamePowerOnData && hasSameFci;
176}
